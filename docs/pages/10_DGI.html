Now for the second stage, let’s assume first that our genotypes ar independent, meaning that \(\mathbf{G} = \mathbf{I}_{v} \otimes\mathbf{\Sigma}_g\), where \(\mathbf{G}\) is the \(vn \times nv\) additive genetic variance matrix, \(\mathbf{\Sigma}_g\) is the \(n \times n\) variance-covariance matrix of traits, \(v\) is the number of genotypes and \(n\) is the number of traits. Let’s build the model and fit the matrices. Let’s fit the following model, where the fixed effect includes just the overall mean and the genotypes are modeled as random. For the residuals, we will assume uncorrelated errors between traits with different variance for each one.

mod_sommer <- mmes(
  fixed = mmean ~ Trait,
  random = ~ vsm(usm(Trait), ism(GID)),
  rcov = ~ vsm(dsm(Trait), ism(units)),
  data = mmeans
)

Let’s reconstruct the matrices needed to obtain the model estimates. Particularly, we are interested in the the genotypic, phenotypic and residual variance covariance matrices \(\mathbf{G}\), \(\mathbf{R}\) and \(\mathbf{V}\) of dimensions \(vn \times nv\), \(\eta \times \eta\) and \(\eta \times \eta\), where \(\eta\) is the number of observations. Please, note that in this particular case \(\eta = v\), and thus the number of records equals the number of genotypes

v <- n_distinct(mmeans$GID)
n <- n_distinct(mmeans$Trait)
eta <- v
sigma_g <- mod_sommer$theta$`vsm(usm(Trait), ism(GID`
sigma_e <- mod_sommer$theta$units
#cov2cor(sigma_g);cov2cor(sigma_e)

I <- diag(nrow = v)
Z <- model.matrix(mmean ~ -1 + GID:Trait, data = mmeans)
G <- I %x% sigma_g
R <- diag(eta) %x% sigma_e
V <- Z %*% G %*% t(Z) + R

Now we know that

\[ \mathbf{G}_{nv \times nv} = \mathbf{I}_{v \times v} \otimes \mathbf{\Sigma}_{n \times n} = \mathbf{I}_{500} \otimes \begin{bmatrix} 0.11 & 0.18 \\ 0.18 & 8.32 \\ \end{bmatrix} = \mathbf{G}_{1000 \times 1000} \]

\[ \mathbf{R}_{\eta \times \eta} = \mathbf{I}_{\eta \times \eta} \otimes \mathbf{\Sigma}_{n \times n} = \mathbf{I}_{500} \otimes \begin{bmatrix} 0.09 & 0 \\ 0 & 17.55 \\ \end{bmatrix} = \mathbf{R}_{1000 \times 1000} \]

\[ \mathbf{V}_{\eta \times \eta} = \mathbf{Z}_{\eta \times nv}\mathbf{G}_{nv \times nv}\mathbf{Z'}_{nv \times \eta} + \mathbf{R}_{\eta \times \eta} = \mathbf{V}_{1000\times 1000} \]

Now, let’s build the Coefficient Matrix as well and obtain the genotypic BLUPs

\[ \begin{equation} \begin{bmatrix} \mathbf{X}'\mathbf{R}^{-1}\mathbf{X} & \mathbf{X}'\mathbf{R}^{-1}\mathbf{Z} \\ \mathbf{Z}'\mathbf{R}^{-1}\mathbf{X} & \mathbf{Z}'\mathbf{R}^{-1}\mathbf{Z} + \mathbf{G}^{-1} \end{bmatrix} \begin{bmatrix} BLUE(\boldsymbol{\beta}) \\ BLUP(\mathbf{u}) \end{bmatrix} = \begin{bmatrix} \mathbf{X}'\mathbf{R}^{-1}\mathbf{y} \\ \mathbf{Z}'\mathbf{R}^{-1}\mathbf{y} \end{bmatrix} \end{equation} ; \begin{bmatrix} BLUE(\boldsymbol{\beta}) \\ BLUP(\mathbf{u}) \end{bmatrix} = C^{-1} \begin{bmatrix} \mathbf{X}'\mathbf{R}^{-1}\mathbf{y} \\ \mathbf{Z}'\mathbf{R}^{-1}\mathbf{y} \end{bmatrix} \]

# Build matrices
y <- as.matrix(mmeans[, "mmean"]) |> na.omit()
X <- model.matrix(mmean ~ -1 + Trait, data = mmeans)
V_inv <- solve(V)
P <- V_inv - V_inv %*% X %*% (solve(t(X) %*% V_inv %*% X)) %*% t(X) %*% V_inv

# Build Coefficient matrix 
C11 <- t(X) %*% solve(R) %*% X
C12 <- t(X) %*% solve(R) %*% Z
C21 <- t(Z) %*% solve(R) %*% X
C22 <- t(Z) %*% solve(R) %*% Z + solve(G)
C <- rbind(
  cbind(C11, C12),
  cbind(C21, C22)
)
C_inv <- solve(C)

# Build Right Hand side matrix
rhs <- as.matrix(rbind(
  t(X) %*% solve(R) %*% y,
  t(Z) %*% solve(R) %*% y
))

# Obtain and arrange BLUPs
solution <- C_inv %*% rhs
u <- data.frame(blup = solution[grep('GID', rownames(solution)), ])
u <- u |>
  rownames_to_column('GID') |>
  mutate(
    GID = gsub('GIDGID', 'GID', GID),
    Trait = gsub('.*:Trait', '', GID),
    GID = gsub(':.*', '', GID)

  pivot_wider(names_from = 'Trait', values_from = 'blup') |>

The last thing we will need to compute the Desire Gain Index is the mean marginal variance covaraince matrix of BLUPs \(\bar{M} = \sum_{i=1}^v \mathbf{M}_{ii}/v\)

M_list <- list()
for (i in 1:v) {
  idx <- (1:n) + n * (i - 1)
  M_list[[i]] <- M[idx, idx] / v
}

M_bar <- Reduce("+", M_list)
M_bar_inv <- solve(M_bar)

Finally, we will compute the Index for all individuals as

\[ \mathbf{I}_{DG} = (\mathbf{d'}\bar{M}^{-1} \otimes \mathbf{I}_v)\mathbf{\hat{g}} \]

Therefore, we can simulate which 20 individuals would be selected based on the index given different desired gaimn values

ghat <- as.matrix(as.vector(as.matrix(u)))
results <- data.frame()
d_list <- list(c(1, 0), c(0, 1), c(1, 1))

for(i in seq_along(d_list)){
  d <- matrix(d_list[[i]], ncol = 1)
  I_dg <- (t(d) %*% M_bar_inv %x% I) %*% ghat
  tmp <- data.frame(
    Index = as.numeric(I_dg)
  )
  
  tmp <- tmp |>
    mutate(
      Selection = ifelse(rank(-Index) <= 100, "yes", "no"),
      dvalues = paste0("GY:", d[1], " Height:", d[2]), 
      Method = 'Werner et al'
    )
  tmp <- cbind(u, tmp)
  results <- rbind(results, tmp)
}

p1 <- ggplot(results, aes(x = GY, y = Height, color = Selection)) +

Additionally, we can build the vector of selection index coefficients as \(\mathbf{\hat{b}}_{DG} = \mathbf{G}^{-1}\mathbf{d}\) so that \(\mathbf{I}_{DG} = \mathbf{b}'\mathbf{y} = \mathbf{d'}\mathbf{G}^{-1} \mathbf{y}\)
